\subsection{Lemma: $\forall n \in \mathbb{N}$, $\exists k \in \mathbb{N}: n = 2k$ or $n =2k + 1$}
		\begin{center}
		\begin{tabular}{c|c}
			case $n = 0$&case $n = 1$\\
			$0 = 2\cdot 0$&$1 = 0 + 1$\\
			&$1 = 2\cdot 0 + 1$\\
			Let $k  =0$& Let $k = 0$\\
			$0 = 2k$&$1 = 2k + 1$
		\end{tabular}
		\end{center}
			Assume $n - 1 = 2k$, or $n - 1 = 2k + 1$\\
		\begin{center}
		\begin{tabular}{c|c}
			case $n - 1 = 2k$&case $n - 1 = 2k + 1$\\
			$n = 2k + 1$&$n = 2k + 2$\\
			&$n = 2(k + 1)$\\
			&Let $k' = k + 1$\\
			&$n = 2k'$
		\end{tabular}
		\end{center}
